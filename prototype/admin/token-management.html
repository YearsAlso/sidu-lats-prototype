Python 示例
import requests

TOKEN = "cam-001-a3f8...7d2c"
URL = "https://lats.example.com/api/v1/detect"

payload = {
    "device_id": "SIDU-CAM-001",
    "timestamp": "2026-07-01T15:42:00Z",
    "image_hash": "sha256:abc123...",
    "defect_count": 3,
    "confidence": 0.94,
    "result": "PASS"
}

headers = {
    "Authorization": f"Bearer {TOKEN}",
    "Content-Type": "application/json"
}

resp = requests.post(URL, json=payload, headers=headers)
print(resp.status_code, resp.json())
